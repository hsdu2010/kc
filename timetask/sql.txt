CREATE TABLE `tm_schedule_job` (
  `id` varchar(36) NOT NULL COMMENT '主键',
  `task_name` varchar(50) NOT NULL COMMENT '任务名称',
  `task_group` varchar(50) NOT NULL COMMENT '任务分组',
  `task_status` varchar(10) NOT NULL COMMENT '任务状态 0禁用 1启用 2删除',
  `cron_expression` varchar(100) NOT NULL COMMENT '任务运行时间表达式',
  `concurrent` varchar(10) NOT NULL COMMENT '是否并发启动任务 0禁用 1启用',
  `remark` varchar(50) DEFAULT NULL COMMENT '描述信息',
  `target_object` varchar(50) NOT NULL COMMENT '执行任务的类/spring注入的id',
  `target_method` varchar(50) NOT NULL COMMENT '任务类中的方法',
  `spring_bean_flag` varchar(10) NOT NULL COMMENT '是否是Spring中定义的Bean 0不是 1是 如果是0需要设置全类名,测试CLAZZ字段需要配置',
  `target_clazz` varchar(500) DEFAULT NULL COMMENT '如果不是Spring中的Bean需要配置全类名用于反射',
  `create_tm` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
  `modify_tm` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '修改时间',
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='定时任务配置表';
